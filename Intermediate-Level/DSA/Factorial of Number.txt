// Code for finding factorial in CPP
#include<iostream>
using namespace std;

int main()
{
    int n;
    cout<<"Enter the number of which you want to find factorial:";
    cin>>n;
    int factorial=1;
    for (int i = 1; i<=n; i++)
    {
        factorial=factorial*i;
    }
    cout<<"Factorial of "<<n<<" is "<<factorial;
    return 0;
}
