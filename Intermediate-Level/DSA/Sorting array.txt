// Write program to sort array using stack
// C
#include <stdio.h>

void bubbleSort(int lst[], int n) {

    for (int i = 0; i < n - 1; ++i) {

        for (int j = 0; j < n - i - 1; ++j) {
            if (lst[j] > lst[j + 1]) {
            int temp = lst[j];
            lst[j] = lst[j + 1];
            lst[j + 1] = temp;
            }
        }
    }
}


void printlst(int lst[], int n) {
    for (int i = 0; i < n; ++i) {
        printf("%d  ", lst[i]);
        }
        printf("\n");
}

int main() {
    int data[] = {-2, 45, 0, 11, -9};

    int n = sizeof(data) / sizeof(data[0]);

    bubbleSort(data, n);

    printf("lst in Ascending Order:\n");
    printlst(data, n);
}
