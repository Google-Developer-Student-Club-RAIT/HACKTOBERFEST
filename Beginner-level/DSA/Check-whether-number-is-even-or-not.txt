//C program for checking whether a number is odd or even
#include<stdio.h>
void main()
{
  int a;
  printf("Enter the number:");
  scanf("%d",&a);
  if(a%2==0)
  printf("THE NUMBER %d IS EVEN",a);
  else
  printf("THE NUMBER %d IS NOT EVEN",a);
 }
