Print the summation of the series:
1! + 2! + 3!/2! + 4!/3! +...
Input: n
Output: sum

#Example:
n=5
Output: 15

// Enter your code here.
// Program in c
#include<stdio.h>
int main()
{
int n,temp,a=1,ans=0;
printf("Enter number of digits desires in the series:");
scanf("%d",&n);
temp=n;
printf("\nThe series is:");
for(n=1;n<=temp;n++)
{
ans=ans+a;
a++;
}
printf("%d",ans);
}
